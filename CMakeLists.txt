CMAKE_MINIMUM_REQUIRED(VERSION 3.2)
PROJECT(gameio)

INCLUDE_DIRECTORIES(${Boost_INCLUDE_DIR})

SET (CMAKE_CXX_STANDARD 11)
SET(SOURCES
	src/Connection.cpp
	src/Server.cpp
)


# tempor√°rio

TARGET_LINK_LIBRARIES(gameclient ${Boost_SYSTEM_LIBRARY})
